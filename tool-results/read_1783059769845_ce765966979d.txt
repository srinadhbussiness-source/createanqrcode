    71→  const { data: analyticsData, isLoading: analyticsLoading } = useQuery({
    72→    queryKey: ['qr-codes', id, 'analytics', range],
    73→    queryFn: () => api.get<AnalyticsSummary>(`/api/qr-codes/${id}/analytics?range=${range}`),
    74→    enabled: !!id && !!code?.isDynamic && plan !== 'free',
    75→    refetchInterval: 15000, // real-ish-time: refresh every 15s
    76→  })
    77→
    78→  const [editingDest, setEditingDest] = useState(false)
    79→  const [destValue, setDestValue] = useState('')
    80→  const [editingPw, setEditingPw] = useState(false)
    81→  const [pwValue, setPwValue] = useState('')
    82→  const [editingExpiry, setEditingExpiry] = useState(false)
    83→  const [expiryValue, setExpiryValue] = useState('')
    84→  const [editingMax, setEditingMax] = useState(false)
    85→  const [maxValue, setMaxValue] = useState('')
    86→  const [editingActivates, setEditingActivates] = useState(false)
    87→  const [activatesValue, setActivatesValue] = useState('')
    88→  const [editingTitle, setEditingTitle] = useState(false)
    89→  const [titleValue, setTitleValue] = useState('')
    90→  const [tagInput, setTagInput] = useState('')
    91→  const [folderDialogOpen, setFolderDialogOpen] = useState(false)
    92→  const [shareDialogOpen, setShareDialogOpen] = useState(false)
    93→
    94→  // Redirect rules (Pro+ feature)
    95→  const [rules, setRules] = useState<RedirectRule[]>(code?.redirectRules ?? [])
    96→  const [newRuleType, setNewRuleType] = useState<'country' | 'device'>('device')
    97→  const [newRuleValue, setNewRuleValue] = useState('')
    98→  const [newRuleDest, setNewRuleDest] = useState('')
    99→
   100→  const patchMut = useMutation({
   101→    mutationFn: (patch: Record<string, unknown>) => api.patch<QrCodeRecord>(`/api/qr-codes/${id}`, patch),
   102→    onSuccess: () => {
   103→      qc.invalidateQueries({ queryKey: ['qr-codes'] })
   104→      qc.invalidateQueries({ queryKey: ['qr-codes', id] })
   105→    },
   106→    onError: () => toast.error('Update failed'),
   107→  })

   109→    mutationFn: () => api.post('/api/seed', { qrCodeId: id }),
   110→    onSuccess: () => {
   111→      toast.success('Demo scans seeded')
   112→      qc.invalidateQueries({ queryKey: ['qr-codes', id, 'analytics'] })
   113→      qc.invalidateQueries({ queryKey: ['qr-codes'] })
   114→      qc.invalidateQueries({ queryKey: ['qr-codes', id] })
   115→    },
   116→    onError: (e) => {
   117→      const msg = e instanceof ApiError ? e.message : 'Seeding failed'
   118→      toast.error(msg)
   119→    },
   120→  })
   121→
   122→  if (isLoading || !code) {
   123→    return (
   124→      <div className="flex h-64 items-center justify-center">
   125→        <Loader2 className="h-6 w-6 animate-spin text-brand" />
   126→      </div>
   127→    )
   128→  }
   129→
   130→  const typeDef = QR_TYPE_MAP[code.qrType]
   131→  const TypeIcon = typeDef?.icon ?? QrCodeIcon
   132→  const shortUrl = code.shortCode ? `createanqrcode.com/q/${code.shortCode}` : ''
   133→  const folders = foldersData?.data ?? []
   134→  const folder = folders.find((f) => f.id === code.folderId)
   135→  const summary = analyticsData
   136→  const top = summary?.topCountries?.[0]
   137→  // Scheduled-activation flag: when activatesAt is set and is in the future,
   138→  // the QR is not yet live and we show a "Scheduled" badge next to the status.
   139→  const isScheduled = !!code.activatesAt && new Date(code.activatesAt) > new Date()
   140→
   141→  // Range options are plan-gated: free=7d, starter=30d, pro=365d, business=unlimited.
   142→  const maxRange = limits.analyticsDays ?? 99999
   143→  const isRangeAllowed = (r: RangeKey) => {
   144→    const opt = RANGE_OPTIONS.find((o) => o.key === r)!
   145→    return opt.days <= maxRange
   146→  }
   147→
   148→  async function handleDownload(kind: 'png' | 'svg' | 'pdf') {
   149→    try {
   150→      const payload = code.staticPayload || code.destinationUrl || ''
   151→      const filename = code.title.replace(/[^a-z0-9-_]+/gi, '_')
   152→      if (kind === 'png') await downloadQrPng(payload, code.design, code.logoDataUrl, filename)
   153→      else if (kind === 'svg') await downloadQrSvg(payload, code.design, code.logoDataUrl, filename)
   154→      else await downloadQrPdf(payload, code.design, code.logoDataUrl, filename, code.title)
   155→      toast.success(`Downloading ${kind.toUpperCase()}`)
   156→    } catch {
   157→      toast.error('Download failed')
   158→    }
   159→  }
   160→
   161→  function copyShort() {
   162→    if (!shortUrl) return
   163→    navigator.clipboard?.writeText(`https://${shortUrl}`)
   164→    toast.success('Short URL copied')
   165→  }
   166→
   167→  function saveDest() {
   168→    if (!destValue.trim()) return
   169→    patchMut.mutate({ destinationUrl: destValue.trim() }, {
   170→      onSuccess: () => { toast.success('Destination updated'); setEditingDest(false) },
   171→    })
   172→  }
   173→  function savePw() {
   174→    patchMut.mutate({ password: pwValue || '' }, {
   175→      onSuccess: () => { toast.success(pwValue ? 'Password set' : 'Password removed'); setEditingPw(false); setPwValue('') },
   176→    })
   177→  }
   178→  function saveExpiry() {
   179→    patchMut.mutate({ expiresAt: expiryValue ? new Date(expiryValue).toISOString() : null }, {
   180→      onSuccess: () => { toast.success('Expiry updated'); setEditingExpiry(false) },
   181→    })
   182→  }
   183→  function saveActivates() {
   184→    patchMut.mutate({ activatesAt: activatesValue ? new Date(activatesValue).toISOString() : null }, {
   185→      onSuccess: () => { toast.success('Activation date updated'); setEditingActivates(false) },
   186→    })
   187→  }
   188→  function saveMax() {
   189→    patchMut.mutate({ maxScans: maxValue ? Number(maxValue) : null }, {
   190→      onSuccess: () => { toast.success('Max scans updated'); setEditingMax(false) },
   191→    })
   192→  }
   193→  function saveRules(next: RedirectRule[]) {
   194→    setRules(next)
   195→    patchMut.mutate({ redirectRules: next.length ? next : null }, {
   196→      onSuccess: () => toast.success(next.length ? 'Redirect rules saved' : 'Redirect rules cleared'),
   197→    })
   198→  }
   199→  function addRule() {
   200→    const val = newRuleValue.trim()
   201→    const dest = newRuleDest.trim()
   202→    if (!val || !dest) { toast.error('Fill in the rule value and destination'); return }
   203→    if (!/^https?:\/\//i.test(dest)) { toast.error('Destination must start with http:// or https://'); return }
   204→    const rule: RedirectRule = { id: `r${Date.now()}`, type: newRuleType, value: val, destination: dest }
   205→    saveRules([...rules, rule])
   206→    setNewRuleValue(''); setNewRuleDest('')
   207→  }
   208→  function removeRule(rid: string) {
   209→    saveRules(rules.filter((r) => r.id !== rid))
   210→  }
   211→  function addTag() {
   212→    const t = tagInput.trim()
   213→    if (!t) return
   214→    if (code.tags.includes(t)) { setTagInput(''); return }
   215→    patchMut.mutate({ tags: [...code.tags, t] }, {
   216→      onSuccess: () => { setTagInput('') },
   217→    })
   218→  }
   219→  function removeTag(t: string) {
   220→    patchMut.mutate({ tags: code.tags.filter((x) => x !== t) })
   221→  }
   222→  function setFolder(folderId: string | null) {
   223→    patchMut.mutate({ folderId }, {
   224→      onSuccess: () => { toast.success('Folder updated'); setFolderDialogOpen(false) },
   225→    })
   226→  }
   227→  function toggleStatus() {
   228→    const next = code.status === 'paused' ? 'active' : 'paused'
   229→    patchMut.mutate({ status: next }, {
   230→      onSuccess: () => toast.success(next === 'active' ? 'QR code resumed' : 'QR code paused'),
   231→    })
   232→  }
   233→  function convertToDynamic() {
   234→    patchMut.mutate({ isDynamic: true }, {
   235→      onSuccess: () => toast.success('Converted to dynamic — analytics enabled!'),
   236→      onError: () => toast.error('Convert failed'),
   237→    })
   238→  }
   239→  function saveTitle() {
   240→    const t = titleValue.trim()
   241→    if (!t) { setEditingTitle(false); return }
   242→    patchMut.mutate({ title: t }, {
   243→      onSuccess: () => { toast.success('Title updated'); setEditingTitle(false) },
   244→    })
   245→  }
   246→
   247→  // Export this QR code's scans as CSV (Pro+). Fetches the full scan list
   248→  // (up to 500 rows) rather than just the 50-row recent feed so the export
   249→  // is genuinely useful for reporting.
   250→  async function handleExportCSV() {
   251→    try {
   252→      const res = await api.get<{ data: ScanRecord[] }>(`/api/qr-codes/${id}/scans`)
   253→      const rows = res.data
   254→      const headers = ['scanned_at', 'country', 'country_code', 'city', 'device', 'os', 'browser', 'referrer', 'language']
   255→      const lines = [headers.join(',')]
   256→      for (const r of rows) {
   257→        const cells = [
   258→          r.scannedAt,
   259→          r.countryName ?? '',
   260→          r.countryCode ?? '',
   261→          r.city ?? '',
   262→          r.deviceType ?? '',
   263→          r.os ?? '',
   264→          r.browser ?? '',
   265→          r.referrer ?? '',
   266→          r.language ?? '',
   267→        ].map((c) => `"${String(c).replace(/"/g, '""')}"`)
   268→        lines.push(cells.join(','))
   269→      }
   270→      const csv = lines.join('\n')
   271→      const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' })
   272→      const url = URL.createObjectURL(blob)
   273→      const a = document.createElement('a')
   274→      a.href = url
   275→      a.download = `${code.title.replace(/[^a-z0-9-_]+/gi, '_')}-scans-${range}.csv`
   276→      a.click()
   277→      URL.revokeObjectURL(url)
   278→      toast.success(`Exported ${rows.length} scans`)
   279→    } catch {
   280→      toast.error('Export failed')
   281→    }
   282→  }
   283→
   284→  return (
   285→    <div className="space-y-6 p-4 sm:p-6 lg:p-8">
   286→      {/* Header */}
   287→      <div className="flex flex-col gap-3 sm:flex-row sm:items-center sm:justify-between">
   288→        <div className="flex min-w-0 items-center gap-3">
   289→          <Button variant="ghost" size="icon" onClick={() => navigate('qr-codes')}>
   290→            <ArrowLeft className="h-4 w-4" />
   291→          </Button>
   292→          <div className="min-w-0">
   293→            <div className="flex items-center gap-2">

   296→                  <Input
   297→                    value={titleValue}
   298→                    onChange={(e) => setTitleValue(e.target.value)}
   299→                    onKeyDown={(e) => { if (e.key === 'Enter') { e.preventDefault(); saveTitle() } if (e.key === 'Escape') setEditingTitle(false) }}
   300→                    onBlur={saveTitle}
   301→                    autoFocus
   302→                    className="h-9 w-64 text-lg font-bold"
   303→                  />
   304→                  <Check className="h-4 w-4 text-brand" />
   305→                </div>
   306→              ) : (
   307→                <>
   308→                  <h1 className="truncate text-xl font-bold tracking-tight sm:text-2xl">{code.title}</h1>
   309→                  <button
   310→                    onClick={() => { setTitleValue(code.title); setEditingTitle(true) }}

   312→                    aria-label="Rename"
   313→                    title="Rename"

   315→                    <Pencil className="h-3.5 w-3.5" />

   321→                className="rounded-md p-1 hover:bg-accent"
   322→                aria-label="Toggle favorite"

   324→                <Star className={cn('h-4 w-4', code.favorite ? 'fill-foreground text-foreground' : 'text-muted-foreground')} />
   325→              </button>
   326→            </div>
   327→            <div className="mt-1 flex flex-wrap items-center gap-1.5">
   328→              <Badge variant="secondary" className="gap-1 text-[10px]">
   329→                <TypeIcon className="h-2.5 w-2.5" /> {typeDef?.label ?? code.qrType}
   330→              </Badge>
   331→              <StatusBadge status={code.status} />
   332→              {code.isDynamic && (
   333→                <Badge variant="outline" className="border-brand/40 text-[10px] text-brand">Dynamic</Badge>
   334→              )}
   335→              {isScheduled && (
   336→                <Badge variant="outline" className="border-brand/40 text-[10px] text-brand">
   337→                  <Clock className="mr-1 h-2.5 w-2.5" /> Scheduled
   338→                </Badge>
   339→              )}
   340→            </div>
   341→          </div>
   342→        </div>
   343→        <div className="flex items-center gap-2">
   344→          <Button variant="outline" onClick={() => setShareDialogOpen(true)}>
   345→            <Share2 className="mr-2 h-4 w-4" /> Share
   346→          </Button>
   347→          <Button onClick={() => openStudioEdit(code.id)} className="bg-brand text-brand-foreground hover:bg-brand/90">
   348→            <Pencil className="mr-2 h-4 w-4" /> Edit QR Code
   349→          </Button>
   350→        </div>
   351→      </div>
   352→
   353→      <div className="grid grid-cols-1 gap-6 lg:grid-cols-3">
   354→        {/* Left: preview + downloads */}
   355→        <Card className="border-border lg:col-span-1">
   356→          <CardHeader><CardTitle className="text-base">QR Code</CardTitle></CardHeader>
   357→          <CardContent className="flex flex-col items-center gap-4">
   358→            <div className="rounded-xl bg-card p-4 shadow-sm ring-1 ring-border">
   359→              <QrPreview
   360→                payload={code.staticPayload || code.destinationUrl || ''}
   361→                design={code.design}
   362→                logoDataUrl={code.logoDataUrl}
   363→                size={280}
   364→              />
   365→            </div>
   366→            <div className="flex w-full flex-col gap-2 sm:flex-row">
   367→              <Button onClick={() => handleDownload('png')} className="flex-1 bg-brand text-brand-foreground hover:bg-brand/90">
   368→                <Download className="mr-2 h-4 w-4" /> PNG
   369→              </Button>
   370→              <Button onClick={() => handleDownload('svg')} variant="outline" className="flex-1">
   371→                <Download className="mr-2 h-4 w-4" /> SVG
   372→              </Button>
   373→              <Button onClick={() => handleDownload('pdf')} variant="outline" className="flex-1">
   374→                <Download className="mr-2 h-4 w-4" /> PDF
   375→              </Button>
   376→            </div>
   377→          </CardContent>
   378→        </Card>
   379→
   380→        {/* Center: scan simulation preview */}
   381→        <Card className="border-border lg:col-span-1">
   382→          <CardHeader>
   383→            <CardTitle className="text-base">Scan Preview</CardTitle>
   384→          </CardHeader>
   385→          <CardContent>
   386→            <ScanPreview code={code} />
   387→          </CardContent>
   388→        </Card>
   389→
   390→        {/* Right: details */}
   391→        <div className="space-y-6 lg:col-span-1">
   392→          <Card className="border-border">
   393→            <CardHeader><CardTitle className="text-base">Details</CardTitle></CardHeader>
   394→            <CardContent className="space-y-3 text-sm">
   395→              <DetailRow icon={Calendar} label="Created" value={formatDateTime(code.createdAt)} />
   396→              <DetailRow icon={Clock} label="Last edited" value={formatDateTime(code.updatedAt)} />
   397→              <DetailRow icon={QrCodeIcon} label="Type" value={typeDef?.label ?? code.qrType} />
   398→              <DetailRow icon={ShieldCheck} label="Error correction" value={code.design.errorCorrection} />
   399→
   400→              {/* Tags */}
   401→              <div className="flex items-start gap-3 pt-1">
   402→                <Tag className="mt-0.5 h-4 w-4 shrink-0 text-muted-foreground" />
   403→                <div className="min-w-0 flex-1">
   404→                  <p className="text-xs font-medium text-muted-foreground">Tags</p>
   405→                  <div className="mt-1 flex flex-wrap items-center gap-1.5">
   406→                    {code.tags.map((t) => (
   407→                      <span key={t} className="inline-flex items-center gap-1 rounded bg-muted px-2 py-0.5 text-xs">
   408→                        #{t}
   409→                        <button onClick={() => removeTag(t)} className="text-muted-foreground hover:text-destructive">
   410→                          <X className="h-3 w-3" />
   411→                        </button>
   412→                      </span>
   413→                    ))}
   414→                    {code.tags.length === 0 && <span className="text-xs text-muted-foreground">No tags yet</span>}
   415→                    <Input
   416→                      value={tagInput}
   417→                      onChange={(e) => setTagInput(e.target.value)}
   418→                      onKeyDown={(e) => { if (e.key === 'Enter') { e.preventDefault(); addTag() } }}
   419→                      placeholder="+ add tag"
   420→                      className="h-7 w-28 text-xs"
   421→                    />
   422→                  </div>
   423→                </div>
   424→              </div>
   425→
   426→              {/* Folder */}
   427→              <div className="flex items-center gap-3 pt-1">
   428→                <FolderIcon className="h-4 w-4 shrink-0 text-muted-foreground" />
   429→                <div className="min-w-0 flex-1">
   430→                  <p className="text-xs font-medium text-muted-foreground">Folder</p>
   431→                  {folder ? (
   432→                    <div className="mt-1 flex items-center gap-2">
   433→                      <span className="h-2.5 w-2.5 rounded-full" style={{ background: folder.color }} />
   434→                      <span className="text-sm">{folder.name}</span>
   435→                    </div>
   436→                  ) : (
   437→                    <p className="mt-0.5 text-sm text-muted-foreground">Unfiled</p>
   438→                  )}
   439→                </div>
   440→                <Button variant="ghost" size="sm" onClick={() => setFolderDialogOpen(true)}>
   441→                  Change
   442→                </Button>
   443→              </div>
   444→            </CardContent>
   445→          </Card>
   446→
   447→          {/* Dynamic details */}
   448→          {code.isDynamic && (
   449→            <Card className="border-brand/30">
   450→              <CardHeader>
   451→                <CardTitle className="flex items-center gap-2 text-base">
   452→                  <Sparkles className="h-4 w-4 text-brand" />
   453→                  Dynamic Code Details
   454→                </CardTitle>
   455→              </CardHeader>
   456→              <CardContent className="space-y-4 text-sm">
   457→                {/* Short URL */}
   458→                <div className="space-y-1.5">
   459→                  <p className="text-xs font-medium text-muted-foreground">Short URL</p>
   460→                  <div className="flex flex-wrap items-center gap-2">
   461→                    <code className="flex-1 truncate rounded-md border border-border bg-muted px-2 py-1.5 text-xs">
   462→                      {shortUrl}
   463→                    </code>
   464→                    <Button variant="outline" size="icon" className="h-8 w-8" onClick={copyShort}>
   465→                      <Copy className="h-3.5 w-3.5" />
   466→                    </Button>

   992→            <button
   993→              onClick={() => setFolder(null)}
   994→              className="flex w-full items-center justify-between rounded-md border border-border px-3 py-2 text-sm hover:bg-accent"
   995→            >
   996→              <span>No folder (unfile)</span>
   997→              {!code.folderId && <Check className="h-4 w-4 text-brand" />}
   998→            </button>
   999→            {folders.map((f) => (
  1000→              <button
  1001→                key={f.id}

  1009→            ))}
  1010→          </div>
  1011→        </DialogContent>
  1012→      </Dialog>
  1013→
  1014→      {/* Share dialog */}
  1015→      <ShareDialog open={shareDialogOpen} onOpenChange={setShareDialogOpen} code={code} />
  1016→    </div>
  1017→  )
